## 1.[面试题 01.08. 零矩阵](https://leetcode-cn.com/problems/zero-matrix-lcci/)

```js
/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 * 思路：
 *  - 使用另外二个数组存储要清零行和列，然后迭代清零即可。
 */
var setZeroes = function(matrix) {
    if(matrix.length ===0 ){
        return matrix;
    }
    let cleanRow  = new Array(matrix.length);
    let cleanColumn = new Array(matrix.length);

    for(let i=0; i<matrix.length; i++){
        for(let j =0; j<matrix[0].length; j++){
            if(matrix[i][j] === 0){
                cleanRow[i] = true;
                cleanColumn[j] = true;
            }
        }
    }

    // 清除行
    for(let i=0; i<matrix.length; i++){
        if(cleanRow[i]){
            matrix[i] = new Array(matrix[0].length).fill(0)
        }
    }

    // 清除列

    for(let j =0; j<matrix[0].length; j++){
        if(cleanColumn[j]){
            for(let i =0 ; i<matrix.length; i++){
                matrix[i][j]=0;
            }
        }
    }

    
};
```

```js
/**
 * @param {number[][]} matrix
 * @return {void} Do not return anything, modify matrix in-place instead.
 */
var setZeroes = function(matrix) {
    let r = matrix.length;
    let c = matrix[0].length;
    
    let rows = {};
    let cols = {};

    for(let i = 0; i < r; i ++)
        for(let j = 0; j < c; j ++) {
            if(matrix[i][j] == 0) {
                rows[i] = true;
                cols[j] = true;
            }
        }
    for(let i = 0; i < r; i ++)
        for(let j = 0; j < c; j ++) {
            if(rows[i]|| cols[j]) {
                matrix[i][j] = 0;
            }
        }
    return matrix;
};
```

```js
 */
var setZeroes = function(matrix) {
  const row_map = {}, col_map = {};
  for(let row = 0; row < matrix.length; row++){
    for(let col = 0; col < matrix[row].length; col++){
      if(matrix[row][col] === 0){
        row_map[row] = true;
        col_map[col] = true;
      }
    } 
  }
  for(let row = 0; row < matrix.length; row++){
    for(let col = 0; col < matrix[row].length; col++){
      if(row_map[row] || col_map[col]){
        matrix[row][col] = 0
      }
    } 
  }

};
```
